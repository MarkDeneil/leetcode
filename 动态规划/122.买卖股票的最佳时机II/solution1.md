本题与 121.买卖股票的最佳时机的区别在于，股票可以买卖多次。因此若第 i-1 天不持有股票，第 i 天持有股票，第 i-1 天有可能是之前持有股票又卖掉的情况，此时已经存在了一定利润，可以把利润加上。  
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        // 本题与 121.买卖股票的最佳时机的最大区别在于：第 121. 题中股票全程只能买一次；本题中股票可以买多次了，但最多同时只能拥有一支股票
        // dp[i][0] 表示第 i 天持有股票能获得的最大利润
        // dp[i][1] 表示第 i 天不持有股票能获得的最大利润
        vector<vector<int>> dp(prices.size(), vector<int>(2, 0));
        dp[0][0] = -prices[0];
        dp[0][1] = 0;
        for (int i = 1; i < prices.size(); ++i) {
            // 第 i 天持有股票，若第 i-1 天持有股票，则 dp[i][0] = dp[i-1][0]，若第 i-1 天未持有股票，因为股票可以买卖多次，第 i 天持有股票可以获得的最大利润就是第 i-1 天前获得的利润减去今天买入股票的钱
            dp[i][0] = max(dp[i-1][0], dp[i-1][1] - prices[i]); // 求解本题与 121. 区别在于此。
            dp[i][1] = max(dp[i-1][1], dp[i-1][0] + prices[i]);
        }
        return dp[prices.size()-1][1];

    }
};
```
